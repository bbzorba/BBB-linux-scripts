netargs=setenv bootargs console=ttyO0,115200n8 root=/dev/mmcblk0p2 rw rootfstype=ext4 rootwait debug earlyprintk

load mmc 0:1 ${loadaddr} uImage; \
load mmc 0:1 ${fdtaddr} am335x-boneblack.dtb; \
run netargs; \
bootm ${loadaddr} - ${fdtaddr}
